\begin{question}[6 points]
        Consider the function
        \[
                f(x) =
                \begin{cases}
                        6x^2 \qquad & \mbox{ if $x<0$}\\
                        -7x-1 \qquad & \mbox{ if $x\geq 0$}
                \end{cases}
        \]
        \begin{enumerate}[(a)]
        \item Show that $f(x)$ is one-to-one.
                \vfill
                \vfill
        \item Determine the inverse function $f^{-1}(x)$.
                \vfill
                \vfill
        \end{enumerate}        
\end{question}
