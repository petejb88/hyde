\begin{exercise}
        Let $f(x)$ be the function
        \[
                f(x) = 
                \begin{cases}
                        2x-1 \qquad & \mbox{if $x<-1$}\\
                        cx & \mbox{if $-1 \leq x \leq 2$}\\
                        |x-2| & \mbox{if $x > 2$}.
                \end{cases}
        \]
        \begin{enumerate}[(a)]
        \item Find the value of $c$ that makes $f(x)$ left-continuous.
                \vfill
        \item Find the value of $c$ that makes $f(x)$ right-continuous.
                \vfill
        \end{enumerate}
\end{exercise}
