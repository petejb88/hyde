\begin{exercise}
        If a ball is thrown vertically upward with a veocity of 80 ft/s, then its height after $t$ seconds is
        $s(t) = 80t - 16t^2$.
        \begin{enumerate}[(a)]
        \item On what closed time interval $[a,b]$ is the ball at or above ground level?
        \item Use the Extreme Value Theorem to compute the maximum height of the ball.
        \item What is the velocity of the ball when it is 96 ft above the ground on its way up? On its way down?
        \end{enumerate}
\end{exercise}

\begin{exercise}
        Consider the function $g(x) = \dfrac{x^2}{(x-2)^2}$.
        \begin{enumerate}[(a)]
        \item Use calculus to find the critical points of $g(x)$.
        \item Use calculus to find the intervals of increase and decrease for $g(x)$.
        \item Use the First Derivative Test to classify each critical point as a
                local max, local min, or neither.
        \end{enumerate}
\end{exercise}
